num = eval(input("Ingrese un número entero de 4 cifras:\n"))
primerDigito = num//1000
segundoDigito = (num//100)%10
tercerDigito = (num//10)%10
cuartoDigito = (num//1)%10
numeroCompleto = str(cuartoDigito)+str(tercerDigito)+str(segundoDigito)+str(primerDigito)
numeroInvertido = int(numeroCompleto)
print("El número invertido al ingresar",num,"es",numeroInvertido)
